Wykorzystaj informację o tym, że $\mathcal F\{\Pi(t)\}=Sa\left(\frac{\omega}{2}\right)$.}{Compute the Fourier transform of the $f(t)=Sa\left(\omega_0 \cdot t\right) \cdot sin\left(\omega_0 \cdot t\right)$ signal using theorems describing the properties of Fourier transformation. Exploit the following transform $\mathcal F\{\Pi(t)\}=Sa\left(\frac{\omega}{2}\right)$.}

\begin{equation}
f(t) = Sa\left(\omega_0 \cdot t\right) \cdot sin\left(\omega_0 \cdot t\right)
\end{equation}

\TT{W pierwszej kolejności można funkcję $f(t)$ rozpisać następująco:}{Firstly, note that using Euler's identity we can write:}

\begin{align*}
f(t) &= Sa\left(\omega_0 \cdot t\right) \cdot sin\left(\omega_0 \cdot t\right)=\\
&=\begin{Bmatrix}
\EulerSin
\end{Bmatrix}=\\
&= Sa\left(\omega_0 \cdot t\right) \cdot \frac{e^{\jmath \cdot \omega_0 \cdot t} - e^{-\jmath \cdot \omega_0 \cdot t}}{2 \cdot \jmath}=\\
&= \frac{1}{2 \cdot \jmath} \cdot \left( Sa\left(\omega_0 \cdot t\right) \cdot e^{\jmath \cdot \omega_0 \cdot t} - Sa\left(\omega_0 \cdot t\right) \cdot e^{-\jmath \cdot \omega_0 \cdot t} \right)=\\
&=\begin{Bmatrix}
f_1(t) &= Sa\left(\omega_0 \cdot t\right) \cdot e^{\jmath \cdot \omega_0 \cdot t} \\
f_2(t) &= Sa\left(\omega_0 \cdot t\right) \cdot e^{-\jmath \cdot \omega_0 \cdot t}
\end{Bmatrix}=\\
&=\frac{1}{2 \cdot \jmath} \cdot \left(f_1(t) - f_2(t) \right)
\end{align*}

\TT{Należy zauważyć iż funkcja $f_1(t)$ i $f_2(t)$ jest złożeniem funkcji $Sa$ i funkcji wykładniczych.}{Please note, that $f_1(t)$ and $f_2(t)$ signals are product of $Sa$ and complex exponential functions:}

\begin{align*}
f_1(t) &= Sa\left(\omega_0 \cdot t\right) \cdot e^{\jmath \cdot \omega_0 \cdot t} = g(t)\cdot e^{\jmath \cdot \omega_0 \cdot t}\\
f_2(t) &= Sa\left(\omega_0 \cdot t\right) \cdot e^{-\jmath \cdot \omega_0 \cdot t} = g(t) \cdot e^{-\jmath \cdot \omega_0 \cdot t}
\end{align*}

\TT{Znając transformatę sygnału $g(t) = Sa\left(\omega_0 \cdot t\right)$ możemy skorzystać z twierdzenia o przesunięciu w dziedzinie częstotliwości.}{Knowing the Fourier transform of $g(t) = Sa\left(\omega_0 \cdot t\right)$ we can use the modulation theorem to derive the Fourier transform of $f_x(t)$:}

\begin{align*}
\FrequencyShiftTeorem{g}{G}{f}{F}
\end{align*}

\TT{Aby wyznaczyć transformatę sygnału $g(t)$ możemy skorzystać z twierdzenia o symetrii. Znając transformatę $H(\jmath \omega)$ sygnału $h(t)$ można wyznaczyć transformatę $G(\jmath \omega)$ sygnału $g(t)$:}{We know that $\mathcal F\{\Pi(t)\}=Sa\left(\frac{\omega}{2}\right)$. Based on time-frequency duality theorem we can derive transform for $g(t)$ signal:}

\begin{align*}
\SymetryTeorem{h}{H}{g}{G}
\end{align*}

\TT{Tak wiec zacznijmy od transformaty sygnału prostokątnego $h(t)=\Pi(t)$ i wyznaczymy transformatę funkcji $Sa$.}{Let's start with deriving Fourier transform for $Sa$ signal using $h(t)=\Pi(t)$:}

\begin{align*}
h(t)=\Pi(t) &\overset{F}{\rightarrow} H(\jmath \omega) = Sa\left(\frac{\omega}{2}\right)\\
g_1(t) = H(t) = Sa\left(\frac{t}{2}\right) &\overset{F}{\rightarrow} 
G_1(\jmath \omega) = 2\pi \cdot h(- \jmath \omega) = \pi \cdot  \Pi\left(-\omega\right) = 2\pi \cdot  \Pi\left(\omega\right)
\end{align*}

\TT{Wyznaczyliśmy transformatę funkcji $g_1(t)$. Jednak funkcja $g_1(t)$ nie ma takiej samej postaci jak funkcja $g(t)$.}{We derived Fourier transform for $g_1(t)$ signal. However, this is not exact the same function as $g(t)$.}

\begin{align*}
g(t)&=Sa\left(\omega_0 \cdot t\right)=\\
&=Sa\left(\omega_0 \cdot t \cdot \frac{2}{2}\right)=\\
&=Sa\left(2\cdot \omega_0 \cdot \frac{t}{2}\right)=\\
&=Sa\left(\frac{2\cdot \omega_0 \cdot t}{2}\right)=\\
&=\begin{Bmatrix}
a = 2\cdot \omega_0
\end{Bmatrix}=\\
&=Sa\left(\frac{a\cdot t}{2}\right)=\\
&=g_1(a\cdot t) 
\end{align*}

\TT{Znając transformatę funkcji $g_1(t)$ możemy wyznaczyć transformatę funkcji $g(t)=g_1(a \cdot t)$ za pomocą twierdzenia o zmianie skali.}{Now, we can calculate the Fourier transform for $g(t)=g_1(a \cdot t)$ signal using the scaling theorem:}

\begin{align*}
\TimeScalingTeorem{g_1}{G_1}{g}{G}
\end{align*}

\begin{align*}
G(\jmath \omega) &= \frac{1}{\left|\alpha \right|} \cdot G_1(\jmath \frac{\omega}{\alpha})=\\
&=\begin{Bmatrix}
\alpha = 2\cdot \omega_0
\end{Bmatrix}=\\
&=\frac{1}{\left| 2\cdot \omega_0 \right|} \cdot G_1( \frac{\omega}{2\cdot \omega_0})=\\
&=\begin{Bmatrix}
G_1(\jmath \omega) = 2\pi \cdot \Pi\left(\omega\right)
\end{Bmatrix}=\\
&=\frac{1}{ 2\cdot \omega_0 } \cdot 2\pi \cdot \Pi\left( \frac{\omega}{2\cdot \omega_0}\right)=\\
&=\frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega}{2\cdot \omega_0}\right)
\end{align*}

\TT{Tak wiec transformata sygnału $g(t)=Sa\left(\omega_0 \cdot t\right)$ jest równa $G(\jmath \omega)=\frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega}{2\cdot \omega_0}\right)$}{As a result the Fourier transform for $g(t)=Sa\left(\omega_0 \cdot t\right)$ signal is equal to $G(\jmath \omega)=\frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega}{2\cdot \omega_0}\right)$.}

\TT{Kolejnym krokiem jest wyznaczenie transformaty dwóch sygnałów}{Now, we are able to derive transforms for:}

\begin{align*}
f_1(t)&=Sa\left(\omega_0 \cdot t\right) \cdot e^{\jmath \cdot \omega_0 \cdot t}\\
f_2(t)&=Sa\left(\omega_0 \cdot t\right) \cdot e^{-\jmath \cdot \omega_0 \cdot t}\\
\end{align*}

\TT{Korzystając z twierdzenie o przesunięciu w dziedzinie częstotliwości}{Based on the modulation theorem we get:}

\begin{align*}
\FrequencyShiftTeorem{g}{G}{f_1}{F_1}
\end{align*}

\begin{align*}
F_1(\jmath \omega)&=G\left(\jmath \left(\omega -\omega_0\right)\right)=\\
&=\frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega - \omega_0}{2\cdot \omega_0}\right)
\end{align*}

\begin{align*}
F_2(\jmath \omega)&=G\left(\jmath \left(\omega +\omega_0\right)\right)=\\
&=\frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega + \omega_0}{2\cdot \omega_0}\right)
\end{align*}

\TT{Ostatecznie korzystając z liniowości transformaty Fouriera}{Finally, based on the linearity theorem}

\begin{align*}
\HomogeneousTeorem{f}{F}
\end{align*}

\TT{otrzymujemy:}{we get:}

\begin{align*}
F(\jmath \omega)&=F_1(\jmath \omega)-F_2(\jmath \omega)=\\
&=\frac{1}{2 \cdot \jmath} \cdot \left( \frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega - \omega_0}{2\cdot \omega_0}\right) - \frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega + \omega_0}{2\cdot \omega_0}\right) \right)
\end{align*}


\TT{Transformata Fouriera sygnału $f(t)=Sa\left(\omega_0 \cdot t\right) \cdot sin\left(\omega_0 \cdot t\right)$ jest równa $F(\jmath \omega)=\frac{1}{2 \cdot \jmath} \cdot \left( \frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega - \omega_0}{2\cdot \omega_0}\right) - \frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega + \omega_0}{2\cdot \omega_0}\right) \right)$}{The Fourier transform of $f(t)=Sa\left(\omega_0 \cdot t\right) \cdot sin\left(\omega_0 \cdot t\right)$ is equal to $F(\jmath \omega)=\frac{1}{2 \cdot \jmath} \cdot \left( \frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega - \omega_0}{2\cdot \omega_0}\right) - \frac{\pi}{ \omega_0 } \cdot \Pi\left( \frac{\omega + \omega_0}{2\cdot \omega_0}\right) \right)$.}
